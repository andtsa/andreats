1use anyhow::Result;
2use gourd_lib::config::slurm::ResourceLimits;
3use gourd_lib::experiment::Experiment;
4use gourd_lib::experiment::FieldRef;
5use gourd_lib::experiment::Run;
6use gourd_lib::experiment::RunInput;
7use gourd_lib::file_system::FileOperations;
8/// This function will generate a new run.
9///
10/// This should be used by all code paths adding runs to the experiment.
11/// This does *not* set the parent and child.
12#[allow(clippy::too_many_arguments)]
13pub fn generate_new_run(
14    run_id: usize,
15    program: usize,
16    run_input: RunInput,
17    input: Option<FieldRef>,
18    input_group: Option<String>,
19    limits: ResourceLimits,
20    parent: Option<usize>,
21    experiment: &Experiment,
22    fs: &impl FileOperations,
23) -> Result<Run> {
24    let seq = experiment.seq;
25    Ok(Run {
26        program,
27        input: run_input,
28        err_path: fs.truncate_and_canonicalize(
29            &experiment
30                .output_folder
31                .join(format!("{seq}/{program}/{run_id}/stderr")),
32        )?,
33        metrics_path: fs.truncate_and_canonicalize(
34            &experiment
35                .metrics_folder
36                .join(format!("{seq}/{program}/{run_id}/metrics")),
37        )?,
38        output_path: fs.truncate_and_canonicalize(
39            &experiment
40                .output_folder
41                .join(format!("{seq}/{program}/{run_id}/stdout")),
42        )?,
43        work_dir: fs.truncate_and_canonicalize_folder(
44            &experiment
45                .output_folder
46                .join(format!("{seq}/{program}/{run_id}/")),
47        )?,
48        afterscript_output: None,
49        limits,
50        slurm_id: None,
51        rerun: None,
52        generated_from_input: input,
53        parent,
54        group: input_group,
55    })
56}

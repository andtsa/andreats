12/// Mapping of the expected architecture to the string used by `lipo`.
13const OSX_ARCH_MAPPING: for<'a> fn(&'a str) -> &'static str = |machine| match machine {
14    "x86" => "i386",
15    "x86_64" => "x86_64",
16    "aarch64" => "arm64",
17    "powerpc" => "ppc",
18    _ => "unsupported",
19};
20
21/// Verifies that the file present at `binary` matches the CPU architecture
22/// provided, specifically for macOS systems. Uses the command-line utility
23/// `lipo` to determine what architecture(s) the binary runs on. If `lipo`
24/// cannot be called, such as on Macs running OS X/PowerPC that have not
25/// received software updates since 2005, the architecture verification is
26/// skipped.
27pub(crate) fn verify_arch(
28    binary: &PathBuf,
29    expected_arch: &str,
30    fs: &impl FileOperations,
31) -> Result<()> {
32    let mut bytes = fs
33        .read_bytes(binary)
34        .context("Could not read the binary file.")?
35        .into_iter();
36
37    // We *DO* allow shebangs.
38    if let Some(0x23) = bytes.next() {
39        if let Some(0x21) = bytes.next() {
40            return Ok(());
41        }
42    }
43
44    match Command::new("lipo")
45        .arg("-archs")
46        .arg(binary.as_path().as_os_str())
47        .output()
48    {
49        Ok(out) => {
50            let binary_archs =
51                String::from_utf8(out.stdout.to_ascii_lowercase()).with_context(ctx!(
52                    "Could not get the output of 'lipo' when checking the binary's architecture.", ;
53                    "",
54                ))?;
55            // `lipo` can return multiple architectures (macOS Universal Binary)
56            if !binary_archs.contains(OSX_ARCH_MAPPING(expected_arch)) {
57                bailc!(
58                  "The program architecture(s) {binary_archs} do not match \
59                    the expected architecture {expected_arch}", ;
60                  "The architecture does not match for program {binary:?}", ;
61                  "Ensure that the program is compiled for the correct target",
62                );
63            }
64            Ok(())
65        }
66        Err(_) => Ok(()),
67    }
68}

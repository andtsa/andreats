17/// The status of a single run
18#[derive(Ord, PartialOrd, Eq, PartialEq, Clone)]
19pub(super) enum RerunStatus {
20    /// Run has not finished yet
21    NotFinished,
22
23    /// Finished successfully, with exit code 0
24    FinishedExitZero,
25
26    /// Finished successfully, and the assigned label has rerun_by_default set
27    /// to false
28    FinishedSuccessLabel(String),
29
30    /// Failed because the assigned label has rerun_by_default set to true
31    FailedErrorLabel(String),
32
33    /// Failed with an exit code
34    FailedExitCode(i32),
35}
36
37impl Display for RerunStatus {
38    fn fmt(&self, f: &mut Formatter<'_>) -> std::fmt::Result {
39        match self {
40            RerunStatus::NotFinished => write!(f, "Not finished"),
41            RerunStatus::FinishedExitZero => write!(f, "Finished with exit code 0"),
42            RerunStatus::FinishedSuccessLabel(l) => write!(f, "Finished with label {l}"),
43            RerunStatus::FailedErrorLabel(l) => write!(f, "Failed with label {l}"),
44            RerunStatus::FailedExitCode(c) => write!(f, "Failed with exit code {c}"),
45        }
46    }
47}

10/// Print user readable infomation about the scheduling status.
11pub fn print_scheduling(exp: &Experiment, starting: bool) -> Result<()> {
12    if starting {
13        info!(
14            "Starting out experiment {PRIMARY_STYLE}{}{PRIMARY_STYLE:#}...",
15            exp.seq
16        );
17    } else {
18        info!(
19            "Continuing experiment {PRIMARY_STYLE}{}{PRIMARY_STYLE:#}...",
20            exp.seq
21        );
22    }
23
24    let total_runs: usize = exp.runs.len();
25
26    let total_scheduled: usize = exp.scheduled_nodep();
27
28    info!("There are {total_runs} total runs,");
29    info!(
30        "Out of which {total_scheduled} have been scheduled and {} are still left unscheduled",
31        total_runs - total_scheduled,
32    );
33
34    if total_scheduled == total_runs {
35        info!("Nothing more to schedule");
36        return Ok(());
37    }
38
39    info!("");
40
41    let mut bar = String::default();
42    let percentage: f64 = (total_scheduled as f64) / (total_runs as f64);
43    let filled = f64::ceil(percentage * (SCHEDULE_BAR_WIDTH as f64)) as usize;
44    let unfilled = SCHEDULE_BAR_WIDTH - filled;
45
46    bar.push_str(&format!("  [{PRIMARY_STYLE}"));
47
48    for _ in 0..filled {
49        bar.push('#');
50    }
51
52    bar.push_str(&format!("{PRIMARY_STYLE:#}{TERTIARY_STYLE}"));
53
54    for _ in 0..unfilled {
55        bar.push('.');
56    }
57
58    bar.push_str(&format!("{TERTIARY_STYLE:#}]"));
59
60    let mut bar_lower = String::default();
61
62    bar_lower.push_str(&format!("   {PRIMARY_STYLE}"));
63
64    for _ in 0..filled {
65        bar_lower.push('^');
66    }
67
68    bar_lower.push_str(&format!(
69        " these ones are already sent to slurm{PRIMARY_STYLE:#}"
70    ));
71
72    info!("{bar}");
73
74    if total_scheduled > 0 {
75        info!("{bar_lower}");
76    }
77
78    info!("");
79    info!("To schedule the rest when this part finishes run {PRIMARY_STYLE}gourd continue{PRIMARY_STYLE:#}");
80
81    Ok(())
82}

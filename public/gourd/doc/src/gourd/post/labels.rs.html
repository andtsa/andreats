1use anyhow::Result;
2use gourd_lib::experiment::Experiment;
3use log::debug;
4use log::trace;
5use log::warn;
6
7/// Assigns a label to a run.
8pub fn assign_label(
9    run_id: usize,
10    source_text: &str,
11    experiment: &Experiment,
12) -> Result<Option<String>> {
13    debug!("Assigning label for text {source_text:?}");
14
15    let mut result_label: Option<String> = None;
16
17    let label_map = &experiment.labels;
18    let mut keys = label_map.keys().collect::<Vec<&String>>();
19    keys.sort_unstable_by(|a, b| label_map[*b].priority.cmp(&label_map[*a].priority));
20
21    for l in keys {
22        let label = &label_map[l];
23        if label.regex.is_match(source_text) {
24            if let Some(ref r) = result_label {
25                warn!("The afterscript for run {run_id:?} matches multiple labels: {r} and {l}");
26            } else {
27                trace!("{source_text} matches {l}");
28                result_label = Some(l.clone());
29            }
30        }
31    }
32
33    Ok(result_label)
34}

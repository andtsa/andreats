16/// Stores a template for `gourd init`: a named set of directory contents.
17#[derive(Debug, Clone)]
18pub struct InitExample<'a> {
19    /// The template name.
20    pub name: &'a str,
21
22    /// The template description.
23    pub description: &'a str,
24
25    /// A tar-gz encoded version of the directory structure
26    /// of the given template/example, containing all files
27    /// within the example directory excluding `gourd.toml`.
28    pub directory_tarball: &'a [u8],
29}
30
31impl InitExample<'_> {
32    /// Extracts the template's tarball in the provided directory.
33    ///
34    /// The directory must have a valid parent, but may not exist.
35    /// This is to be enforced by the caller method.
36    pub fn unpack_to(&self, directory: &Path, file_system: &FileSystemInteractor) -> Result<()> {
37        let tar = GzDecoder::new(self.directory_tarball);
38        let mut archive = Archive::new(tar);
39
40        // Do not preserve the creation time, etc. when unpacking the template.
41        archive.set_preserve_mtime(false);
42
43        debug!("Unpacking the example archive");
44        file_system.write_archive(directory, archive)?;
45
46        if !file_system.dry_run {
47            debug!("Entering the directory {directory:?}");
48            let previous_dir = std::env::current_dir()?;
49            std::env::set_current_dir(directory)?;
50
51            let config_path = PathBuf::from("gourd.toml");
52
53            debug!("Checking for a \"gourd.toml\" at {config_path:?}.");
54            match Config::from_file(Path::new("gourd.toml"), file_system) {
55                Err(e) => {
56                    debug!("Configuration check failed: {}", e.root_cause());
57                    warn!(
58                        "The \"gourd.toml\" configuration in this example is missing or invalid."
59                    );
60                    warn!("You may have to make some changes.");
61                }
62                Ok(_) => debug!("A valid \"gourd.toml\" is present."),
63            }
64
65            debug!("Returning to the previous directory {:?}", &previous_dir);
66            std::env::set_current_dir(previous_dir)?;
67        }
68
69        Ok(())
70    }
71}

73/// Retrieves all available examples.

105/// Retrieves a named example experiment, or [None].
106pub fn get_example(id_input: &str) -> Option<InitExample<'static>> {
107    let id = id_input.to_string().replace(['.', '_', ' '], "-");
108    debug!("Translating the example-id: {id_input} to {id}");
109    get_examples().get(&id as &str).cloned()

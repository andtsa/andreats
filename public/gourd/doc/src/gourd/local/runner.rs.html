14use log::trace;

16/// Run a list of tasks locally in a multithreaded way.
17pub async fn run_locally(
18    tasks: Vec<Command>,
19    force: bool,
20    sequential: bool,
21    mut num_threads: usize,
22) -> Result<()> {
23    if tasks.len() > TASK_LIMIT && !force && !sequential {
24        bailc!(
25          "task limit exceeded", ;
26          "{PRIMARY_STYLE}gourd{PRIMARY_STYLE:#} will not run more than \
27          {TASK_LIMIT} jobs on local, doing so may possibly exhaust your file descriptors", ;
28          "if you are {NAME_STYLE}absolutely{NAME_STYLE:#} sure that you \
29          want to run {} tasks use the {PRIMARY_STYLE}--force{PRIMARY_STYLE:#} \
30          option", tasks.len()
31        )
32    }
33
34    #[cfg(not(tarpaulin_include))] // Tarpaulin can't calculate the coverage correctly
35    tokio::spawn(async move {
36        /// Error in case of wrapper failure.
37        fn handle_output(join: io::Result<Output>) {
38            if let Ok(exit) = join {
39                if !exit.status.success() {
40                    error!("Failed to run gourd wrapper: {:?}", exit.status);
41                    error!(
42                        "Wrapper returned: {}",
43                        String::from_utf8(exit.stderr).unwrap()
44                    );
45                    process::exit(1);
46                }
47            } else {
48                error!("Couldn't start the wrapper: {join:?}");
49                error!("Ensure that the wrapper is accessible. (see man gourd)");
50                process::exit(1);
51            }
52        }
53
54        if sequential {
55            for mut task in tasks {
56                trace!("Running task: {task:?}");
57                handle_output(task.output());
58            }
59        } else {
60            // Buffering 0 tasks will prevent anything from happening.
61            // We use 0 to indicate no upper limit. See documentation
62            if num_threads == 0 {
63                num_threads = usize::MAX;
64            }
65
66            let handles = tokio_stream::iter(tasks)
67                .map(|mut task| {
68                    trace!("Queueing task: {task:?}");
69                    tokio::task::spawn_blocking(move || task.output())
70                })
71                // only poll up to `num_threads` of tasks at once:
72                .buffer_unordered(num_threads);
73
74            tokio::pin!(handles);
75            while let Some(join_result) = handles.next().await {
76                match join_result {
77                    Ok(output) => handle_output(output),
78                    Err(join_error) => {
79                        error!(
80                            "Could not join the child in the multithreaded runtime: {join_error}"
81                        );
82                        process::exit(1);
83                    }
84                }
85            }
86        }
87
88        Result::<()>::Ok(())
89    });
90
91    Ok(())
92}

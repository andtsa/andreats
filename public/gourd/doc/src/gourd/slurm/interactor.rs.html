30/// Creates a Slurm duration string.
31///
32/// Converts a standard `std::time::Duration` to a Slurm duration in one of
33/// the following formats: {ss, mm:ss, hh:mm:ss, d-hh:mm:ss}
34pub fn format_slurm_duration(duration: Duration) -> String {
35    let secs = duration.as_secs();
36    let secs_rem = secs % 60;
37
38    if secs == secs_rem {
39        return format!("{secs:0>2}");
40    }
41
42    let mins = secs / 60;
43    let mins_rem = mins % 60;
44
45    if mins == mins_rem {
46        return format!("{mins:0>2}:{secs_rem:0>2}");
47    }
48
49    let hours = mins / 60;
50    let hours_rem = hours % 24;
51
52    if hours == hours_rem {
53        return format!("{hours:0>2}:{mins_rem:0>2}:{secs_rem:0>2}");
54    }
55
56    let days = hours / 24;
57
58    format!("{days}-{hours_rem:0>2}:{mins_rem:0>2}:{secs_rem:0>2}")
59}

77/// Get a limit from the `sacctmgr` command.
78fn sacctmgr_limit(field: &str) -> Result<String> {
79    let mut cmd = Command::new("sacctmgr");
80    cmd.arg("show")
81        .arg("user")
82        .arg("withassoc")
83        .arg("where")
84        .arg(format!("name={}", std::env::var("USER")?))
85        .arg(format!("format={field}"))
86        .arg("--parsable2")
87        .arg("--noheader");
88
89    let out = cmd.output()?;
90    debug!("Running {cmd:?} gave {:?}", &out);
91
92    Ok(String::from_utf8_lossy(&out.stdout).trim().to_string())
93}

95/// Get a limit from the `scontrol` command.
96fn scontrol_limit(field: &str) -> Result<String> {
97    let s_control_out = Command::new("scontrol")
98        .arg("show")
99        .arg("config")
100        .output()?
101        .stdout;
102    let s_control_str = String::from_utf8_lossy(&s_control_out);
103    Ok(s_control_str
104        .split('\n')
105        .filter(|x| x.contains(field))
106        .collect::<Vec<&str>>()[0]
107        .split('=')
108        .collect::<Vec<&str>>()[1]
109        .trim()
110        .to_string())
111}

118    fn get_version(&self) -> Result<[u64; 2]> {
119        let s_info_out = Command::new("sinfo").arg("--version").output()?;
120        let version = String::from_utf8_lossy(&s_info_out.stdout)
121            .to_string()
122            .split_whitespace()
123            .collect::<Vec<&str>>()[1]
124            .split(|c: char| !c.is_numeric())
125            .collect::<Vec<&str>>()
126            .iter()
127            .map(|x| x.parse::<u64>().unwrap())
128            .collect::<Vec<u64>>();
129        let mut buf = [0; 2];
130
131        buf[0] = *version.first().ok_or(anyhow!("Invalid version received"))?;
132        buf[1] = *version.get(1).ok_or(anyhow!("Invalid version received"))?;
133
134        Ok(buf)
135    }

137    fn get_partitions(&self) -> Result<Vec<Vec<String>>> {
138        let s_info_out = Command::new("sinfo").arg("-o").arg("%P %a").output()?;
139        let partitions = String::from_utf8_lossy(&s_info_out.stdout)
140            .split('\n')
141            .map(|x| x.to_string())
142            .map(|y| {
143                y.split_whitespace()
144                    .collect::<Vec<&str>>()
145                    .iter()
146                    .map(|z| z.to_string())
147                    .collect::<Vec<String>>()
148            })
149            .collect::<Vec<Vec<String>>>();
150        Ok(partitions)
151    }

169    fn max_submit(&self) -> Result<usize> {
170        let out = Command::new("sacctmgr")
171            .arg("show")
172            .arg("qos")
173            .arg("format=MaxSubmitPU")
174            .arg("--parsable2")
175            .arg("--noheader")
176            .output()?;
177
178        if let Some(max_submit) = String::from_utf8_lossy(&out.stdout)
179            .split_whitespace()
180            .next()
181        {
182            Ok(max_submit.parse()?)
183        } else {
184            match sacctmgr_limit("MaxSubmit")?.parse() {
185                Ok(x) => Ok(x),
186                Err(e) => {
187                    debug!("Could not parse max submissions from slurm: {e}");
188                    Ok(usize::MAX) // since this function is only used for
189                                   // checking limits,
190                                   // if we can't parse the limit then just
191                                   // ignore the check
192                }
193            }
194        }
195    }

207    fn schedule_chunk(
208        &self,
209        slurm_config: &SlurmConfig,
210        chunk: &Chunk,
211        experiment: &mut Experiment,
212        exp_path: &Path,
213    ) -> Result<()> {
214        let resource_limits = chunk.limits();
215        let chunk_index = experiment.register_runs(&chunk.runs);
216
217        let optional_args = parse_optional_args(slurm_config);
218
219        // `%A` gets replaced with array *job* id, `%a` with the array *task* id
220        // this is read in `src/gourd/status/slurm_files.rs` to get the output.
221        let slurm_out = experiment
222            .slurm_out("%A_%a")
223            .ok_or(anyhow!("Slurm config not found (unreachable)"))?;
224        let slurm_err = experiment
225            .slurm_err("%A_%a")
226            .ok_or(anyhow!("Slurm config not found (unreachable)"))?;
227
228        let contents = format!(
229            "#!/bin/bash
230#SBATCH --job-name=\"{}\"
231#SBATCH --array=\"{}-{}\"
232#SBATCH --ntasks=1
233#SBATCH --partition=\"{}\"
234#SBATCH --time=\"{}\"
235#SBATCH --cpus-per-task=\"{}\"
236#SBATCH --mem-per-cpu=\"{}\"
237#SBATCH --account=\"{}\"
238#SBATCH --output={:?}
239#SBATCH --error={:?}
240{}
241set -x
242
243{} {} {} $SLURM_ARRAY_TASK_ID
244",
245            slurm_config.experiment_name,
246            0,
247            &chunk.runs.len() - 1,
248            slurm_config.partition,
249            format_slurm_duration(resource_limits.time_limit),
250            resource_limits.cpus,
251            resource_limits.mem_per_cpu,
252            slurm_config.account,
253            slurm_out,
254            slurm_err,
255            optional_args,
256            experiment.wrapper,
257            exp_path.display(),
258            chunk_index
259        );
260
261        debug!("Sbatch file: {contents}");
262
263        let mut cmd = Command::new("sbatch")
264            .arg("--parsable")
265            .stdin(Stdio::piped())
266            .stdout(Stdio::piped())
267            .spawn()
268            .with_context(ctx!(
269              "Failed to submit batch job to SLURM", ;
270              "Ensure that you have permissions to submit jobs to the cluster",
271            ))?;
272
273        cmd.stdin
274            .as_mut()
275            .ok_or(anyhow!("Could not connect to sbatch"))
276            .context("")?
277            .write_all(contents.as_bytes())
278            .with_context(ctx!(
279                "Failed to submit batch job to SLURM", ;
280                "Tried submitting this script {contents}",
281            ))?;
282
283        trace!("sbatch child: {:?}", &cmd);
284
285        let proc = cmd.wait_with_output().context("")?;
286
287        debug!("sbatch exited with code {:?}", proc.status);
288        trace!("sbatch stdout:\n{:?}", proc.stdout);
289        trace!("sbatch stderr:\n{:?}", proc.stderr);
290
291        if !proc.status.success() {
292            bailc!("Sbatch failed to run", ;
293                "Sbatch printed: {}", String::from_utf8_lossy(&proc.stderr);
294                "Please ensure that you are running on slurm",
295            );
296        }
297
298        let batch_id = String::from_utf8(proc.stdout)
299            .with_context(ctx!(
300              "Could not decode sbatch output", ; "",
301            ))?
302            .trim()
303            .to_string();
304
305        trace!("This chunk was scheduled with id: {batch_id}");
306        experiment.mark_chunk_scheduled(chunk, batch_id);
307
308        Ok(())
309    }

323    fn get_accounting_data(&self, since: &DateTime<Local>) -> Result<Vec<SacctOutput>> {
324        let mut sacct_cmd = Command::new("sacct");
325        sacct_cmd
326            .arg("-p")
327            .arg("--allocations")
328            .arg("--starttime")
329            .arg(since.format("%Y-%m-%d %H:%M:%S").to_string()) // YYYY-MM-DD[THH:MM[:SS]] from slurm docs
330            .arg("--endtime=now")
331            .arg("--format=jobid,jobname,state,exitcode");
332
333        trace!("Gathering slurm status with: {sacct_cmd:?}");
334
335        let sacct = sacct_cmd.output().with_context(ctx!(
336          "Could not get accounting data", ;
337          "Make sure that the `sacct` program is accessible",
338        ))?;
339
340        let mut result = Vec::new();
341
342        for job in String::from_utf8_lossy(&sacct.stdout)
343            .trim()
344            .split('\n')
345            .skip(1)
346        {
347            let fields = job.split('|').collect::<Vec<&str>>();
348            let exit_codes = fields[3].split(':').collect::<Vec<&str>>();
349
350            result.push(SacctOutput {
351                job_id: fields[0].to_string(),
352                job_name: fields[1].to_string(),
353                state: fields[2].to_string(),
354                slurm_exit_code: exit_codes[0].parse().unwrap_or(0),
355                program_exit_code: exit_codes[1].parse().unwrap_or(0),
356            });
357        }
358
359        Ok(result)
360    }

362    fn scheduled_jobs(&self) -> Result<Vec<String>> {
363        let sacct = Command::new("squeue")
364            .arg("--me")
365            .arg("--array")
366            .arg("-h") // hide the table header
367            .arg("--Format=jobid,arraytaskid")
368            .arg("--state=PD,R,CG")
369            .output()
370            .with_context(ctx!(
371              "Could not get scheduled jobs", ;
372              "Make sure that the `squeue` program is accessible",
373            ))?;
374
375        Ok(String::from_utf8_lossy(&sacct.stdout)
376            .lines()
377            .map(|line| {
378                let mut parts = line.split_whitespace();
379                match (parts.next(), parts.next()) {
380                    (Some(job), Some(task)) => format!("{}_{}", job.trim(), task.trim()),
381                    _ => unreachable!("Error in parsing squeue output. This should not happen."),
382                }
383            })
384            .collect())
385    }

403    fn cancel_jobs(&self, batch_ids: Vec<String>) -> Result<()> {
404        if batch_ids.len() < SHORTEN_STATUS_CUTOFF {
405            info!(
406                "Cancelling runs {TERTIARY_STYLE}[{}]{TERTIARY_STYLE:#}",
407                batch_ids.join(", ")
408            );
409        } else {
410            info!("Cancelling {} runs", batch_ids.len());
411        }
412
413        for chunk in batch_ids.chunks(500) {
414            let mut cancel = Command::new("scancel");
415            cancel.args(chunk);
416
417            debug!("Running cancel: {cancel:?}");
418
419            let output = cancel.output().with_context(ctx!(
420              "Failed to cancel runs",;
421              "Make sure that the `scancel` program is accessible",
422            ))?;
423
424            if !output.status.success() {
425                bailc!("Failed to cancel runs", ;
426                    "\"scancel\" printed: {}", String::from_utf8_lossy(&output.stderr);
427                    "",
428                );
429            }
430        }
431
432        info!("{} runs cancelled", batch_ids.len());
433
434        Ok(())
435    }

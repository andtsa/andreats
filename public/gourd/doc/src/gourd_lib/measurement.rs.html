5use std::fmt::Display;
6use std::time::Duration;
7
8use serde::Deserialize;
9use serde::Serialize;
10
11use crate::constants::NAME_STYLE;
12
13/// The metrics of running a program.
14#[derive(Serialize, Deserialize, Debug, Clone, Copy, PartialEq)]
15#[serde(tag = "type")]
16pub enum Metrics {
17    /// The metrics have not been calculated yet.
18    NotCompleted,
19
20    /// The measurement has been finished.
21    Done(Measurement),
22}
23
24/// This structure contains the measurements for one run of the binary.
25#[derive(Serialize, Deserialize, Debug, Clone, Copy, PartialEq)]
26pub struct Measurement {
27    /// Interval of wall time.
28    pub wall_micros: Duration,
29    /// The exit code of the invoked program.
30    pub exit_code: i32,
31    /// The rusage of the invoked program.
32    pub rusage: Option<RUsage>,
33}
34
35/// Resource usage statistics for a process.
36#[derive(Serialize, Deserialize, Debug, Clone, Copy, PartialEq)]
37pub struct RUsage {
38    /// User CPU time used.
39    pub utime: Duration,
40    /// System CPU time used.
41    pub stime: Duration,
42    /// Maximum resident set size.
43    pub maxrss: usize,
44    /// Integral shared memory size.
45    pub ixrss: usize,
46    /// Integral unshared data size.
47    pub idrss: usize,
48    /// Integral unshared stack size.
49    pub isrss: usize,
50    /// Page reclaims (soft page faults).
51    pub minflt: usize,
52    /// Page faults (hard page faults).
53    pub majflt: usize,
54    /// Swaps.
55    pub nswap: usize,
56    /// Block input operations.
57    pub inblock: usize,
58    /// Block output operations.
59    pub oublock: usize,
60    /// IPC messages sent.
61    pub msgsnd: usize,
62    /// IPC messages received.
63    pub msgrcv: usize,
64    /// Signals received.
65    pub nsignals: usize,
66    /// Voluntary context switches.
67    pub nvcsw: usize,
68    /// Involuntary context switches.
69    pub nivcsw: usize,
70}
71
72impl Display for RUsage {
73    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
74        writeln!(
75            f,
76            "  {NAME_STYLE}user  cpu  time{NAME_STYLE:#}: {}",
77            humantime::Duration::from(self.utime)
78        )?;
79        writeln!(
80            f,
81            "  {NAME_STYLE}system cpu time{NAME_STYLE:#}: {}",
82            humantime::Duration::from(self.stime)
83        )?;
84        writeln!(
85            f,
86            "  {NAME_STYLE}page faults{NAME_STYLE:#}: {}",
87            self.majflt
88        )?;
89        writeln!(
90            f,
91            "  {NAME_STYLE}signals received{NAME_STYLE:#}: {}",
92            self.nsignals
93        )?;
94        writeln!(
95            f,
96            "  {NAME_STYLE}context switches{NAME_STYLE:#}: {}",
97            self.nvcsw + self.nivcsw
98        )?;
99
100        Ok(())
101    }
102}

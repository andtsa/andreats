8/// Deserializing duration from a human-readable string.
9pub fn deserialize_human_time_duration<'de, D>(deserializer: D) -> Result<Duration, D::Error>
10where
11    D: Deserializer<'de>,
12{
13    /// The default duration visitor.
14    struct DurationVisitor;
15
16    impl Visitor<'_> for DurationVisitor {
17        type Value = Duration;
18
19        fn expecting(&self, formatter: &mut fmt::Formatter) -> fmt::Result {
20            formatter.write_str("a valid duration string")
21        }
22
23        fn visit_str<E>(self, v: &str) -> Result<Self::Value, E>
24        where
25            E: serde::de::Error,
26        {
27            humantime::parse_duration(v).map_err(|parse_err| {
28                serde::de::Error::custom(format!("This is not a valid duration: {parse_err}"))
29            })
30        }
31    }
32
33    deserializer.deserialize_str(DurationVisitor {})
34}
35
36/// Serialize duration into a human-readable format
37pub fn serialize_duration<S: Serializer>(duration: &Duration, ser: S) -> Result<S::Ok, S::Error> {
38    S::serialize_str(ser, &humantime::format_duration(*duration).to_string())

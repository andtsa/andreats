27/// This will take a path and canonicalize it.
28pub fn canon_path(path: &Path, fs: &impl FileOperations) -> Result<PathBuf> {
29    fs.canonicalize(path)
30        .map_err(|e| {
31            debug!("Canonicalize error: {e:?}");
32            anyhow!(
33                "failed to find {path:?} with workdir {:?}",
34                current_dir().unwrap()
35            )
36        })
37        .with_context(ctx!("",;"",))
38}

40/// Takes the set of all inputs and expands the globbed arguments.
41///
42/// # Examples
43/// ```toml
44/// [inputs.test_input]
45/// arguments = [ "path|/test/**/*.jpg" ]
46/// ```
47///
48/// May get expanded to:
49///
50/// ```toml
51/// [inputs.test_input_glob_0]
52/// arguments = [ "/test/a/a.jpg" ]
53///
54/// [inputs.test_input_glob_1]
55/// arguments = [ "/test/a/b.jpg" ]
56///
57/// [inputs.test_input_glob_2]
58/// arguments = [ "/test/b/b.jpg" ]
59/// ```
60pub fn expand_argument_globs(
61    inputs: &BTreeMap<String, UserInput>,
62    fs: &impl FileOperations,
63) -> Result<BTreeMap<String, UserInput>> {
64    let mut result = BTreeMap::new();
65
66    for (original, input) in inputs {
67        let mut globset = HashSet::new();
68        globset.insert(input.clone());
69
70        let mut is_glob = false;
71
72        for arg_index in 0..input.arguments.len() {
73            let mut next_globset = HashSet::new();
74
75            for input_instance in &globset {
76                is_glob |=
77                    explode_glob_set(input_instance, original, arg_index, &mut next_globset, fs)?;
78            }
79
80            swap(&mut globset, &mut next_globset);
81        }
82
83        if is_glob {
84            for (idx, glob) in globset.iter().enumerate() {
85                result.insert(
86                    format!("{original}{INTERNAL_PREFIX}{INTERNAL_GLOB}{idx}"),
87                    glob.clone(),
88                );
89            }
90        } else {
91            result.insert(original.clone(), input.clone());
92        }
93    }
94
95    Ok(result)
96}

98/// Given a `input` and `arg_index` expand the glob at that
99/// argument and put the results in `fill`.
100fn explode_glob_set(
101    input: &UserInput,
102    input_name: &str, // only used for warnings.
103    arg_index: usize,
104    fill: &mut HashSet<UserInput>,
105    fs: &impl FileOperations,
106) -> Result<bool> {
107    let arg = &input.arguments[arg_index];
108    let no_escape = arg.strip_prefix(GLOB_ESCAPE);
109
110    if let Some(globbed) = no_escape {
111        for path in glob(globbed).map_err(|_| {
112            anyhow!(
113                "could not expand the glob {globbed}, \
114                any arguments starting with `{GLOB_ESCAPE}` are considered globs"
115            )
116        })? {
117            let mut glob_instance = input.clone();
118
119            glob_instance.arguments[arg_index] = canon_path(
120                &path.map_err(|_| anyhow!("the glob failed to evaluate at {no_escape:?}"))?,
121                fs,
122            )?
123            .to_str()
124            .ok_or(anyhow!("failed to stringify {no_escape:?}"))
125            .with_context(ctx!("",;"",))?
126            .to_string();
127
128            fill.insert(glob_instance);
129        }
130
131        Ok(true)
132    } else {
133        if Path::new(arg).iter().count() > 1 {
134            warn!(
135                " \n\
136                It looks like you specified a path argument: \
137                {WARNING_STYLE}{arg}{WARNING_STYLE:#} \
138                in input {PRIMARY_STYLE}{input_name}{PRIMARY_STYLE:#}, \
139                but did not prefix it with {CMD_DOC_STYLE} {GLOB_ESCAPE} {CMD_DOC_STYLE:#}\n\
140                {HELP_STYLE}tip:{HELP_STYLE:#} Consider changing the argument to \
141                {CMD_STYLE}\"{GLOB_ESCAPE}{arg}\"{CMD_STYLE:#} \
142                in order to canonicalize the path and expand any globs.\n\n\
143                {HELP_STYLE}You can safely ignore this warning.{HELP_STYLE:#}\
144            "
145            );
146        }
147        fill.insert(input.clone());
148        Ok(false)
149    }

43/// A pair of a path to a binary and cli arguments.
44#[derive(Serialize, Deserialize, Debug, Clone, PartialEq, Hash, Eq)]
45#[serde(deny_unknown_fields)]
46pub struct UserProgram {
47    /// A path to the executable.
48    pub binary: Option<PathBuf>,
49
50    /// Fetch the program binary remotely
51    /// ### Permissions
52    /// If this file is fetched on unix, the permissions
53    /// for it are: `rwxr-xr--`.
54    pub fetch: Option<FetchedResource<0o754>>,
55
56    /// A git reference to the program.
57    pub git: Option<GitProgram>,
58
59    /// The cli arguments for the executable.
60    #[serde(default = "EMPTY_ARGS")]
61    pub arguments: Vec<String>,
62
63    /// The path to the afterscript, if there is one.
64    ///
65    /// Afterscripts are run after the main program has finished.
66    /// It can be used for a quick postprocess of the main program's output,
67    /// and the afterscript output can be used for labeling the job in `gourd
68    /// status`, or serving as a custom metric in CSV exporting.
69    #[serde(default)]
70    pub afterscript: Option<PathBuf>,
71
72    /// Resource limits to optionally overwrite default resource limits.
73    #[serde(default)]
74    pub resource_limits: Option<ResourceLimits>,
75
76    /// The programs to postprocess this one.
77    #[serde(default)]
78    pub next: Vec<String>,
79}

107/// A pair of a path to an input and additional cli arguments.
108///
109/// # Examples
110///
111/// ```toml
112/// [programs.test_program]
113/// binary = "test"
114/// arguments = [ "a", "b" ]
115///
116/// [inputs.test_input]
117/// arguments = [ "c" ]
118/// ```
119///
120/// Will run `test a b c`
121#[derive(Serialize, Deserialize, Debug, Clone, PartialEq, Hash, Eq)]
122#[serde(deny_unknown_fields)]
123pub struct UserInput {
124    /// Direct path to the input.
125    pub file: Option<PathBuf>,
126
127    /// A glob of input files
128    pub glob: Option<String>,
129
130    /// Fetch the input file remotely
131    /// ### Permissions
132    /// If this file is fetched on unix, the permissions
133    /// for it are: `rw-r--r--`.
134    pub fetch: Option<FetchedResource<0o644>>,
135
136    /// Mark this input as belonging to a specific group of inputs.
137    pub group: Option<String>,
138
139    /// The additional cli arguments for the executable.
140    ///
141    /// ### Default
142    /// By default these will be empty.
143    #[serde(default = "EMPTY_ARGS")]
144    pub arguments: Vec<String>,
145}

166/// A parameter.
167///
168/// # Examples
169///
170/// ```toml
171/// [parameters.x]
172/// values = ["1", "2"]
173///
174/// [parameters.y]
175/// values = ["a", "b"]
176///
177/// [programs.test_program]
178/// binary = "test"
179///
180/// [inputs.test_input]
181/// arguments = [ "param|x" ]
182/// ```
183///
184/// Will run:
185/// `test 1 a`
186/// `test 1 b`
187/// `test 2 a`
188/// `test 2 b`
189#[derive(Debug, Clone, PartialEq, Serialize, Deserialize, Hash, Eq)]
190#[serde(deny_unknown_fields)]
191pub struct Parameter {
192    /// Sub-parameters of this parameter.
193    ///
194    /// To be used exclusively without values of parameter.
195    pub sub: Option<BTreeMap<String, SubParameter>>,
196
197    /// The values of parameter.
198    ///
199    /// To be used exclusively without sub (parameter).
200    pub values: Option<Vec<String>>,
201}
202
203/// A subparameter.
204///
205/// # Examples
206///
207/// ```toml
208/// [parameters.x.sub.a]
209/// values = ["1", "2", "3"]
210///
211/// [parameters.x.sub.b]
212/// values = ["15", "60", "30"]
213///
214/// [programs.test_program]
215/// binary = "test"
216///
217/// [inputs.test_input]
218/// arguments = [ "subparam|x.a", "subparam|x.b" ]
219/// ```
220///
221/// Will run:
222/// `test 1 15`
223/// `test 2 60`
224/// `test 3 30`
225#[derive(Debug, Clone, PartialEq, Serialize, Deserialize, Hash, Eq)]
226#[serde(deny_unknown_fields)]
227pub struct SubParameter {
228    /// The values of sub parameter.
229    ///
230    /// Has to be equal in length to values of other subparameters of the same
231    /// argument.
232    pub values: Vec<String>,
233}

264/// A config struct used throughout the `gourd` application.
265//
266// changing the config struct? see notes in ./tests/config.rs
267// 1. is the change necessary?
268// 2. will it break user workflows?
269// 3. update the tests
270// 4. update the user documentation
271#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
272#[serde(deny_unknown_fields)]
273pub struct Config {
274    // Basic settings.
275    /// The path to a folder where the experiment output will be stored.
276    pub output_path: PathBuf,
277
278    /// The path to a folder where the metrics output will be stored.
279    pub metrics_path: PathBuf,
280
281    /// The path to a folder where the experiments will be stored.
282    pub experiments_folder: PathBuf,
283
284    /// The list of tested algorithms.
285    #[serde(rename = "program")]
286    pub programs: BTreeMap<String, UserProgram>,
287
288    /// The list of inputs for each of them.
289    ///
290    /// The name of an input cannot contain '_i_'.
291    #[serde(rename = "input")]
292    pub inputs: BTreeMap<String, UserInput>,
293
294    /// A path to a TOML file that contains input combinations.
295    pub input_schema: Option<PathBuf>,
296
297    /// The list of parameters.
298    #[serde(rename = "parameter")]
299    pub parameters: Option<BTreeMap<String, Parameter>>,
300
301    /// If running on a SLURM cluster, the job configurations.
302    pub slurm: Option<SlurmConfig>,
303
304    /// If running on a SLURM cluster, the initial global resource limits.
305    pub resource_limits: Option<ResourceLimits>,
306
307    //
308    // Advanced settings.
309    /// The command to execute to get to the wrapper.
310    #[serde(
311        default = "WRAPPER_DEFAULT",
312        skip_serializing_if = "wrapper_is_default"
313    )]
314    pub wrapper: String,
315
316    /// Options for configuring the execution of
317    /// `gourd run local`
318    pub local: Option<LocalOptions>,
319
320    /// Allow custom labels to be assigned based on the afterscript output.
321    ///
322    /// syntax is:
323    /// ```toml
324    /// [labels.<label_name>]
325    /// // the regex where if it matches then this label is assigned
326    /// regex = "<regex>"
327    /// // whether using rerun failed will rerun this job-
328    /// // i.e. is this label a "failure"
329    /// rerun_by_default = true
330    /// ```
331    #[serde(rename = "label")]
332    pub labels: Option<BTreeMap<String, Label>>,
333}

356impl Config {
357    /// Load a `Config` struct instance from a TOML file at the provided path.
358    /// Returns a valid `Config` or an explanatory
359    /// `GourdError::ConfigLoadError`.
360    pub fn from_file(path: &Path, fs: &FileSystemInteractor) -> Result<Config> {
361        let mut initial: Config = fs.try_read_toml(path).with_context(ctx!(
362          "Could not parse {path:?}", ;
363          "More help and examples can be found with \
364          {CMD_STYLE}man gourd.toml{CMD_STYLE:#}",
365        ))?;
366
367        if let Some(schema) = &initial.input_schema {
368            initial.inputs = Config::parse_schema_inputs(schema.as_path(), initial.inputs, fs)?;
369            initial.input_schema = None;
370        }
371
372        Ok(initial)
373    }
374
375    /// Parse the additional inputs toml file and add them to the inputs map.
376    pub fn parse_schema_inputs(
377        path_buf: &Path,
378        mut inputs: BTreeMap<String, UserInput>,
379        fs: &impl FileOperations,
380    ) -> Result<BTreeMap<String, UserInput>> {
381        let hi = fs.try_read_toml::<InputSchema>(path_buf)?;
382
383        for (idx, input) in hi.inputs.iter().enumerate() {
384            inputs.insert(
385                format!("{idx}{INTERNAL_PREFIX}{INTERNAL_SCHEMA_INPUTS}"),
386                input.clone(),
387            );
388        }
389
390        Ok(inputs)
391    }
392}
